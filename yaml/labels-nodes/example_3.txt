# Присвоить две метки указанной ноде и перезаписать (overwrite) значение меток, если они уже существовали или имели другие значения.

kubectl label --overwrite node NODE_NAME disk=ssd gpu=true



# Например, если нода называется Minikube, команда будет такая:

kubectl label --overwrite node minikube disk=ssd gpu=true
